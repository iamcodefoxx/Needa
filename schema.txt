CREATE TABLE `login`.`user` (
  `id` INT UNSIGNED AUTO_INCREMENT,
  `first_name` VARCHAR(30),
  `last_name` VARCHAR(30),
  `email` VARCHAR(100),
  `password` BINARY(60),
  `token` BINARY(20),
  `token_expires` BINARY(13),
  `member_since` VARCHAR(10),
  `status` VARCHAR(8) DEFAULT 'Inactive', 
  `admin` VARCHAR(3) DEFAULT 'No',
  `profile_photo` VARCHAR(50) DEFAULT NULL,
  `cover_photo` VARCHAR(50) DEFAULT NULL,
  `city` VARCHAR(60) DEFAULT NULL,
  `state` CHAR(2) DEFAULT NULL,
  `gender` CHAR(1) DEFAULT NULL,
  `profession` VARCHAR(50) DEFAULT NULL,
  `specialty` TINYTEXT DEFAULT NULL,
  `about` TEXT DEFAULT NULL,
  `skills` TEXT DEFAULT NULL,
  `twitter_profile` TINYTEXT DEFAULT NULL,
  `instagram_profile` TINYTEXT DEFAULT NULL,
  `facebook_profile` TINYTEXT DEFAULT NULL,
  `linkedin_url` TINYTEXT DEFAULT NULL,
  `website_url` TINYTEXT DEFAULT NULL,
  `phone` VARCHAR(11) DEFAULT NULL,
  `display_phone` VARCHAR(3) DEFAULT NULL,
  `display_email` VARCHAR(100) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC) VISIBLE,
  UNIQUE INDEX `email_UNIQUE` (`email` ASC) VISIBLE);
